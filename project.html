3 طبقات حماية

الأمان الحقيقي مش طبقة واحدة. كل طبقة بتحل مشكلة مختلفة، وبعض الطبقات مع بعض بيعملوا درعاً كاملاً ضد الهجمات.

HMAC Authentication

الـ PC والـ Arduino عندهم مفتاح سري مشترك. كل أمر بتبعته بتضيفله "بصمة رقمية" محسوبة من المفتاح ده. لو حد بعت أمر من غير المفتاح الصح، الـ HMAC مش هيتطابق والأمر بيتّرفض.

message: LED_ON|5|839201
HMAC: A3F9B2C1 ← بصمة المفتاح
02
🔢

Sequence Numbers

كل أمر بياخد رقم تسلسلي متزايد (1، 2، 3...). الـ Arduino بيحتفظ بآخر رقم وصله. لو جاه رقم أصغر أو مساوي، معناه إن الأمر قديم أو متكرر فبيتّرفض.

lastSeq = 5
incoming = 3 ← REJECTED (old)
incoming = 6 ← ACCEPTED
03
🎲

Nonce Freshness Window

كل أمر بيتضاف ليه رقم عشوائي فريد (nonce). الـ Arduino بيحتفظ بقائمة آخر 20 nonce شافها. لو نفس الـ nonce جه تاني مرة، حتى لو الـ HMAC صح، الأمر بيتّرفض كـ Replay Attack.

nonce cache: [839201, 445522, ...]
incoming: 839201 ← REPLAY!

✅ فحص HMAC (طبقة 1)

Arduino يحسب HMAC من المفتاح السري بتاعه ويقارنه بالـ HMAC اللي جاي. لو مختلف = مفتاح غلط أو تعديل في الأمر.

expected = HMAC("LED_ON|6|839201", "MySecretKey123")
received = A3F9B2C1
expected == received → PASS ✓

✅ فحص Sequence (طبقة 2)

Arduino يتحقق إن رقم الـ Sequence أكبر من آخر رقم شافه. ده بيمنع إعادة الأوامر القديمة.

incoming = 6
6 > 5 → PASS ✓
lastSeq = 6 ← update

✅ فحص Nonce (طبقة 3)

Arduino يتحقق إن الـ Nonce مش موجود في كاش الـ 20 nonce الأخيرة. لو موجود = Replay Attack.

nonce_cache = [112233, 445566, 778899...]
839201 not in cache → PASS ✓
cache.add(839201) ← store

// Project 108: Command Authentication + Replay Protection
// Arduino Uno + W5500 Ethernet

#include <SPI.h>
#include <Ethernet.h>
#include <EthernetUdp.h>

// ====== Network Config ======
byte mac[] = { 0xDE, 0xAD, 0xBE, 0xEF, 0xFE, 0xED };
IPAddress ip(192, 168, 1, 177);
unsigned int localPort = 8888;

// ====== Shared Secret Key ======
const char SECRET_KEY[] = "MySecretKey123";

// ====== Protocol State ======
uint32_t lastSeq = 0;
uint32_t usedNonces[20];
int nonceIndex = 0;
const int LED_PIN = 7;

EthernetUDP udp;
char packetBuffer[256];

// ====== XOR-HMAC (educational) ======
uint32_t computeHMAC(const char* msg, const char* key) {
  uint32_t hash = 0x12345678;
  int klen = strlen(key);
  int mlen = strlen(msg);
  for (int i = 0; i < mlen; i++) {
    hash ^= (uint32_t)msg[i] << (i % 24);
    hash += (uint32_t)key[i % klen] * 31;
    hash = (hash << 5) | (hash >> 27);
  }
  return hash;
}

bool isNonceUsed(uint32_t nonce) {
  for (int i = 0; i < 20; i++)
    if (usedNonces[i] == nonce) return true;
  return false;
}

void storeNonce(uint32_t nonce) {
  usedNonces[nonceIndex % 20] = nonce;
  nonceIndex++;
}

void setup() {
  Serial.begin(9600);
  pinMode(LED_PIN, OUTPUT);
  Ethernet.begin(mac, ip);
  udp.begin(localPort);
  memset(usedNonces, 0, sizeof(usedNonces));
}

void loop() {
  int pSize = udp.parsePacket();
  if (!pSize) return;
  udp.read(packetBuffer, sizeof(packetBuffer));
  packetBuffer[pSize] = '\0';

  char cmd[32], seqStr[16], nonceStr[16], hmacStr[16];
  if (sscanf(packetBuffer, "%31[^|]|%15[^|]|%15[^|]|%15s",
             cmd, seqStr, nonceStr, hmacStr) != 4) {
    sendReply("ERROR:MALFORMED"); return;
  }

  uint32_t seq      = (uint32_t)atol(seqStr);
  uint32_t nonce    = (uint32_t)atol(nonceStr);
  uint32_t rcvHmac  = (uint32_t)strtoul(hmacStr, NULL, 16);

  // Check 1: HMAC
  char msgToSign[128];
  snprintf(msgToSign, sizeof(msgToSign), "%s|%lu|%lu", cmd, seq, nonce);
  if (computeHMAC(msgToSign, SECRET_KEY) != rcvHmac) {
    sendReply("ERROR:BAD_HMAC"); return;
  }

  // Check 2: Sequence
  if (seq <= lastSeq) {
    sendReply("ERROR:OLD_SEQ"); return;
  }

  // Check 3: Nonce
  if (isNonceUsed(nonce)) {
    sendReply("ERROR:REPLAY"); return;
  }

  // All checks passed!
  lastSeq = seq;
  storeNonce(nonce);

  if (strcmp(cmd, "LED_ON") == 0) {
    digitalWrite(LED_PIN, HIGH);
    sendReply("OK:LED_ON");
  } else if (strcmp(cmd, "LED_OFF") == 0) {
    digitalWrite(LED_PIN, LOW);
    sendReply("OK:LED_OFF");
  }
}

void sendReply(const char* msg) {
  udp.beginPacket(udp.remoteIP(), udp.remotePort());
  udp.write(msg);
  udp.endPacket();
}

import socket, random, time

ARDUINO_IP   = "192.168.1.177"
ARDUINO_PORT = 8888
SECRET_KEY   = "MySecretKey123"

sock = socket.socket(socket.AF_INET, socket.SOCK_DGRAM)
sock.settimeout(2)
seq_counter = 0

def compute_hmac(msg: str, key: str) -> int:
    h = 0x12345678
    for i, ch in enumerate(msg):
        h ^= ord(ch) << (i % 24)
        h += ord(key[i % len(key)]) * 31
        h = ((h << 5) | (h >> 27)) & 0xFFFFFFFF
    return h

def build_packet(cmd, key=SECRET_KEY, seq=None, nonce=None):
    global seq_counter
    if seq is None:
        seq_counter += 1
        seq = seq_counter
    if nonce is None:
        nonce = random.randint(100000, 999999)
    msg = f"{cmd}|{seq}|{nonce}"
    hmac = compute_hmac(msg, key)
    return f"{msg}|{hmac:08X}", seq, nonce

def send_cmd(cmd):
    pkt, _, _ = build_packet(cmd)
    print(f"→ {pkt}")
    sock.sendto(pkt.encode(), (ARDUINO_IP, ARDUINO_PORT))
    try:
        data, _ = sock.recvfrom(256)
        print(f"← {data.decode()}")
    except socket.timeout:
        print("← timeout")

# Send legitimate commands
send_cmd("LED_ON")
time.sleep(1)
send_cmd("LED_OFF")
